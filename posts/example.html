Intro

Syntax highlighting of various languages. 

// Scala comment
def test(a: String): Unit = println(s"Hello, $a")

// D comment
import std.stdio;

void test(string a) {
  writeln("Hello, " ~ a);
}

; Scheme/Racket comment
(define (test a)
  (println (string-append "Hello, " a)))

-- Haskell comment
test :: IO ()
test = println "Hello, world"
